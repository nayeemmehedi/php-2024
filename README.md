// Boolean
if (false) {
    echo "This will not be printed.";
}

// Integer
if (0) {
    echo "This will not be printed.";
}

// Float
if (0.0) {
    echo "This will not be printed.";
}

// String
if ("") {
    echo "This will not be printed.";
}

if ("0") {
    echo "This will not be printed.";
}

// Array
if (array()) {
    echo "This will not be printed.";
}

// NULL
if (NULL) {
    echo "This will not be printed.";
}
